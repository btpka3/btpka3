link:https://maven.apache.org/plugins/maven-toolchains-plugin/index.html[maven-toolchains-plugin]

[source,shell]
----
# 需pom.xml
mvn toolchains:select-jdk-toolchain -Dtoolchain.jdk.version=11

# 无需pom.xml : 列出发现的jdk版本
mvn toolchains:display-discovered-jdk-toolchains

# 无需pom.xml : 把发现的jdk版本 输出成xml
mvn toolchains:generate-jdk-toolchains-xml -Dtoolchain.file=toolchains.xml

# 生成的缓存信息
ls -l ~/.m2/discovered-jdk-toolchains-cache.xml
----

== 示例: toolchains.xml

[source,xml]
----
<?xml version="1.0" encoding="UTF-8"?>
<toolchains xsi:schemaLocation="http://maven.apache.org/TOOLCHAINS/1.1.0 http://maven.apache.org/xsd/toolchains-1.1.0.xsd" xmlns="http://maven.apache.org/TOOLCHAINS/1.1.0"
    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance">
  <toolchain>
    <type>jdk</type>
    <provides>
      <current>true</current>
      <vendor.version>Liberica-NIK-23.1.6-1</vendor.version>
      <vendor>BellSoft</vendor>
      <runtime.version>21.0.6+10-LTS</runtime.version>
      <lts>true</lts>
      <runtime.name>OpenJDK Runtime Environment</runtime.name>
      <env>JAVA_HOME</env>
      <version>21.0.6</version>
    </provides>
    <configuration>
      <jdkHome>/Users/zll/.sdkman/candidates/java/23.1.6.r21-nik</jdkHome>
    </configuration>
  </toolchain>
</toolchains>
----
