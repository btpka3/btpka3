# 显示支持的命令列表。
jcmd $PID help

# 获取 JVM 的版本信息
jcmd $PID VM.version
# 列出 JVM 的运行时参数（包括 JVM 选项和调试标志）。
jcmd $PID VM.flags
# 显示加载的动态库信息。
jcmd $PID VM.dynlibs
# 强制退出 JVM。⚠️
jcmd $PID VM.exit
# 显示 JVM 的系统属性（如 java.version, os.name 等）。
jcmd $PID VM.system_properties
# 解锁商业功能（如 Flight Recorder）。
jcmd $PID VM.unlock_commercial_features
# 触发垃圾回收。
jcmd $PID VM.systemgc
# 获取 JVM 的当前时间（纳秒级）。
jcmd $PID VM.current_time
# 显示 JVM 的运行时长。
jcmd $PID VM.uptime
# 显示 JVM 的启动命令行参数。
jcmd $PID VM.command_line
# 显示 JVM 的锁信息（如死锁检测）。
jcmd $PID VM.monitor_info

# 生成类的直方图（显示对象数量和内存占用）。
jcmd $PID GC.class_histogram filename={file-path}
# 强制执行垃圾回收
jcmd $PID GC.run
# 生成堆转储文件
jcmd $PID GC.heap_dump /path/to/dump.hprof
# 旧版堆转储命令（JDK 8 及更早版本）。
jcmd $PID VM.heap_dump /path/to/dump.hprof
# 运行对象终结器（调用 Object.finalize()）。
jcmd $PID GC.run_finalization
# 旋转垃圾回收日志文件（仅适用于启用 GC 日志的环境）。
jcmd $PID GC.rotate_log

# 获取进程 1234 的线程信息
jcmd $PID Thread.print


jcmd $PID JFR.start
jcmd $PID JFR.check
jcmd $PID JFR.configure
jcmd $PID JFR.dump
jcmd $PID JFR.start
jcmd $PID JFR.stop
jcmd $PID JFR.view
