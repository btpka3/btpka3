




## 使用基于docker的构建环境
* link:https://docs.openharmony.cn/pages/v6.0/zh-cn/OpenHarmony-Overview_zh.md[Docker Environment]
* link:https://github.com/hqzing/dockerharmony[docker-mini-openharmony]
* https://github.com/Harmonybrew

[source,shell]
----
# For mini-system devices:
docker pull swr.cn-south-1.myhuaweicloud.com/openharmony-docker/docker_oh_small:3.2
# For small-system devices:
docker pull swr.cn-south-1.myhuaweicloud.com/openharmony-docker/docker_oh_mini:3.2


# 运行
docker run -it --rm swr.cn-south-1.myhuaweicloud.com/openharmony-docker/docker_oh_small:3.2

docker run --name=ohos -itd --platform linux/arm64 docker.1ms.run/hqzing/docker-mini-openharmony:latest

#docker pull hqzing/dockerharmony:latest
docker pull ghcr.io/hqzing/dockerharmony:latest

skopeo copy \
  --src-no-creds \
  --multi-arch all \
docker://ghcr.io/hqzing/dockerharmony:latest dir:${HOME}/docker-images/dockerharmony

# 👍🏻
skopeo copy \
  --src-no-creds \
  --override-arch arm64 \
  --override-os linux \
  dir:${HOME}/docker-images/dockerharmony \
  docker-archive:${HOME}/docker-images/dockerharmony.tar

podman run -it --rm ghcr.io/hqzing/dockerharmony:latest sh -l
----
