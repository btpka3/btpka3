

## 基本用法
[source,shell]
----


grep
-H, --with-filename   # 输出文件名

grep -l SOME_TO_SEARCH # 只显示出匹配上的文件名

----


=== 排除某些文件名

[source,shell]
----
cd ~
rm -fr /tmp/demo-find
mkdir -p /tmp/demo-find/{a,b,c}
cd /tmp/demo-find

cat > a/a.txt <<EOF
a11 hello a12
a21 hello a22
EOF

cat > b/b.log <<EOF
b11 hello b12
b21 hello b22
EOF

cat > c/c.plain <<EOF
c11 hello c12
c21 hello c22
EOF

# 排除文件 *.log
find .  -type f ! -path "*.log"
# 排除目录 a
find .  -type f ! -path "*/a/*"

----
