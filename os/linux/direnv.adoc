* link:https://direnv.net/[direnv]
* link:https://direnv.net/docs/hook.html[hook]

[source,shell]
----
brew install direnv
# BASH
vi ~/.bashrc
eval "$(direnv hook bash)"

# ZSH  (⚠️ 这里的示例是使用 homebrew 安装的，故下面的 eval 语句需要放到 homebrew 相关初始化语句后面。 )
vi ~/.zshrc
eval "$(direnv hook zsh)"

# Oh my zsh
vi ~/.zshrc
plugins=(... direnv)




----


## 使用的环境变量

[source,shell]
----
# 环境变量
XDG_CONFIG_HOME=$HOME/.config
XDG_DATA_HOME=$HOME/.local/share

# 配置文件
$XDG_CONFIG_HOME/direnv/direnv.toml
$XDG_CONFIG_HOME/direnv/direnvrc
$XDG_CONFIG_HOME/direnv/lib/*.sh            # 方便第三方扩展，在加载 *.envrc 前执行
$XDG_DATA_HOME/direnv/allow
----


## 配置 direnv.toml

 vi ~/.config/direnv/direnv.toml


[source,toml]
----
[global]
# bash_path=
# disable_stdin
## 是否加载 .env , 如果同时存在 .env, .envrc , 则先加载 .envrc
# load_dotenv=true
## Valid time units are “ns”, “us” (or “µs”), “ms”, “s”, “m”, “h”.
# warn_timeout=5s
hide_env_diff=true

## 不输出日志
log_format = "-"

[whitelist]
prefix = [
  "/path1",
  "/path2",
  "/path3"
]
----


## 示例


[source,shell]
----
cd /tmp

rm -fr /tmp/xxx
mkdir -p /tmp/xxx

echo "FOO=${FOO}"

direnv allow /tmp/xxx
echo  "export FOO=foo" > /tmp/xxx/.envrc

cd /tmp/xxx





# 进入目标工程目录
touch .envrc
touch .env


----
