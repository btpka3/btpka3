# 把 `|` 作为分隔符，并输出给定字段。 输出使仍将使用原有分隔符。
cat <<EOF |
aaa|101|102
bbb|201|202
ccc|301|302
EOF
cut -d "|" -f 1,3

# 按字节提取 [1,3] 字节内容
# TIP: `-b1-3` 与 `-b-3` 等同
cat <<EOF |
aaa|101|102
bbb|201|202
ccc|301|302
EOF
cut -b1-3

# 按字节提取 从 第5位 到行尾的内容
cat <<EOF |
aaa|11
bbb|22222
ccc|3333
EOF
cut -b5-
