== 卸载 / uninstall : clean-agency-agents.sh

[source,shell]
----
#!/usr/bin/env bash
set -euo pipefail

# 从 ~/.claude/agents/ 中删除来自 agency-agents 仓库的文件
# 逻辑：遍历 agency-agents 的 agent 定义文件，与 ~/.claude/agents/ 比对
#       - 完全一致：移至系统 Trash
#       - 本地有修改：跳过并提示
#       - 目标不存在：忽略

SRC="${AGENCY_AGENTS_SRC:-$HOME/data0/work/git-repo/github/msitarzewski/agency-agents}"
DST="${CLAUDE_AGENTS_DST:-$HOME/.claude/agents}"
BACKUP_DIR="$HOME/.qoderwork/workspace/moiw7wujddn9dyt7/backup/agency-agents-$(date +%Y%m%d)"

mkdir -p "$BACKUP_DIR"

deleted=0
modified=0
skipped=0

echo "Source: $SRC"
echo "Target: $DST"
echo "Backup: $BACKUP_DIR"
echo "---"

for srcfile in $(find "$SRC" -name "*.md" \
  -not -path "*/.git/*" \
  -not -path "*/.cursor/*" \
  -not -path "*/.github/*" \
  -not -path "*/scripts/*" \
  -not -path "*/.agents/*" \
  -not -path "*/strategy/*" \
  -not -path "*/examples/*" \
  -not -path "*/integrations/*" \
  -not -name "README*" \
  -not -name "CONTRIBUTING*" \
  -not -name "SECURITY*" \
  -not -name "PULL_REQUEST*" \
  -not -name "EXECUTIVE*" \
  -not -name "QUICKSTART*" \
  -not -name "nexus-strategy.md" \
  -not -name "agent-activation*" \
  -not -name "handoff-templates*" \
  -not -name "phase-*" \
  -not -name "scenario-*" \
  2>/dev/null); do

  fname=$(basename "$srcfile")
  target="$DST/$fname"

  if [ ! -f "$target" ]; then
    skipped=$((skipped + 1))
    continue
  fi

  if diff -q "$srcfile" "$target" >/dev/null 2>&1; then
    # 完全一致：备份后删除
    cp "$target" "$BACKUP_DIR/"
    mv -n "$target" ~/.Trash/
    deleted=$((deleted + 1))
    echo "  DELETED: $fname"
  else
    # 本地有修改：保留并提示
    echo "  MODIFIED (skipped): $fname"
    modified=$((modified + 1))
  fi
done

echo "---"
echo "Summary: deleted=$deleted, modified(skipped)=$modified, not-found=$skipped"
echo "Remaining agents: $(ls "$DST" 2>/dev/null | wc -l)"
echo "Backups: $(ls "$BACKUP_DIR" 2>/dev/null | wc -l) files in $BACKUP_DIR"
----
