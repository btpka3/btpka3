* link:https://skills.sh/docs[skills.sh]
* link:https://www.npmjs.com/package/skills[npm-skills]
* link:https://github.com/vercel-labs/skills[github : vercel-labs/skills]


[source,shell]
----

# 请配置全局环境变量（比如 ~/.zshrc ) 禁止上传统计数据
DISABLE_TELEMETRY=1
pnpm add -g skills

skills --help

# 会默认先安装到 ~/.agents/skills/*
# 用 skills 平台注册的名称 安装（用 skills 平台的OpenAPI)
skills add <skill-name>
skills add vercel-labs/agent-skills
# 查找本地目录里的skill并全局安装
skills add ./my-local-skills -g
# 用git代码仓库直接安装，只有能找到合法的 SKILL.md ，无论目录层级。
skills add git@github.com:your-org/your-private-skills.git
# 示例: git仓库特定子目录
skills add github.com/vercel-labs/agent-skills/tree/dev/skills/.curated
# 示例: git仓库特定分支、特定子目录
skills add https://github.com/vercel-labs/agent-skills#main@pr-review
skills add https://github.com/your-org/your-skills.git

skills remove
skills list
skills find [query]
skills update               # 更新安装的skill
# 非交互式安装。
# --copy 表示使用复制模式，如果不使用，则默认使用软连接的方式。
# --all 等同于 `--skill '*' --agent '*' -y`
skills add <source> --agent claude-code --global --all
----
